	is_triggered_only = yes

		ruler_has_mage_personality = yes
		
		#If you dont have any level 1 spells then this will fire
		NOT = { has_ruler_flag = abjuration_1 }
		NOT = { has_ruler_flag = conjuration_1 }

country_event = {
	id =  magic_ruler.2
	title =  magic_ruler.2.t
	desc =  magic_ruler.2.d
	picture = NEW_HEIR_eventPicture
	
	trigger = {
		ruler_has_mage_personality = yes
		has_heir = no
		has_regency = no
		government = monarchy
		is_lesser_in_union = no
		has_government_attribute = heir
		has_consort = yes
		NOT = { ruler_is_immortal = yes }
		NOT = { ruler_age = 60 }
	}
	
	mean_time_to_happen = {
		months = 1000
		modifier = {
			factor = 0.1
			has_global_flag = crimson_deluge
			is_year = 1560
			capital_scope = { superregion = escann_superregion }
		}
		modifier = {
			factor = 0.5
			has_reform = magical_elite_reform
		}
		modifier = {
			factor = 0.25
			OR = {
				ruler_has_mage_personality = yes
				consort_has_mage_personality = yes
			}
		}
	}

	option = {
		name = magic_ruler.2.a
		
		define_heir = {
			age = 0
			dynasty = ROOT
			hidden = yes
		}
		set_heir_mage_effect = yes
	}
}
